cmake_minimum_required(VERSION 3.28)

# Configure this to change the Project's name
set(GAME_NAME "Game")

project(${GAME_NAME} LANGUAGES CXX)

# set for C++ 2017.
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

include(FetchContent)

# SFML repository
FetchContent_Declare(
	SFML
	GIT_REPOSITORY https://github.com/SFML/SFML.git
	GIT_TAG			3.1.0
)

# Enables the solution folder grouping in visual studio
FetchContent_MakeAvailable(SFML)

# collect the project's source and header files located under /source.
file(GLOB_RECURSE SOURCES "source/*.cpp" "source/*.h")

# Automatically reconfigures CMake whenever new files are added to the source folder.
set_property(DIRECTORY APPEND PROPERTY CMAKE_CONFIGURE_DEPENDS "source")

# Mirror source folders tructure to visual studio filters
source_group(TREE ${CMAKE_CURRENT_SOURCE_DIR}/source PREFIX "source" FILES ${SOURCES})

add_executable(${GAME_NAME} ${SOURCES})

# Set working directory to root
set_target_properties(${GAME_NAME} PROPERTIES VS_DEBUGGER_WORKING_DIRECTORY "${CMAKE_CURRENT_SOURCE_DIR}")

# to copy our dlls to the exe's directory.
add_custom_command(TARGET ${GAME_NAME} POST_BUILD
	COMMAND ${CMAKE_COMMAND} -E echo "Skipping DLL copy - using static libs"
)

# to copy our content folder to the exe's directory.
add_custom_command(TARGET ${GAME_NAME} POST_BUILD
	COMMAND ${CMAKE_COMMAND}
	-E copy_directory ${CMAKE_SOURCE_DIR}/content
	$<TARGET_FILE_DIR:${GAME_NAME}>/content
)

# Link against the required SFML modules.
target_link_libraries(${GAME_NAME} PRIVATE
	SFML::Graphics
	SFML::Audio
	SFML::Window
	SFML::System
)

# adds many source directories as an include directory.
target_include_directories(${GAME_NAME} PRIVATE
	source
	source/Bullet
	source/Bullet/Attacks
	source/Bullet/Spawners
	source/Core
	source/Engine
	source/Particle
	source/Player
	source/Sounds
	source/Textures
	source/UI
)

# compiler warnings just in case
if(MSVC)
	target_compile_options(${GAME_NAME} PRIVATE /W4)
else()
	target_compile_options(${GAME_NAME} PRIVATE -Wall -Wextra)
endif()
